import java.util.Scanner;

public class Solution {

	public static void main(String[] args) {
		Scanner scan = new Scanner(System.in);
		int T = scan.nextInt();
		
		for(int i=1; i<=T; i++) {
			int[] time = new int[4];
			int hour   = 0;	// 시간
			int minute = 0;	// 분	
			
			// 시각 2개 입력받기(시, 분, 시, 분)
			for(int j=0; j<time.length; j++) {
				time[j] = scan.nextInt();
			}
			
			// 시의 합이 13 이상일 때 => 합 - 12(13시 => 1시, 18시 => 6시)
			if(time[0] + time[2] >= 13)		hour = time[0] + time[2] - 12;
			else 							hour = time[0] + time[2];	// 시 합
			
			// 분의 합이 60 이상일때 => 시 + 1
			if(time[1] + time[3] >= 60) {
				hour += 1;
				if(hour >= 13)	hour -= 12;
				minute = time[1] + time[3] - 60;
			}
			else {
				minute = time[1] + time[3];
			}
			System.out.println("#" + i + " " + hour + " " + minute);
			scan.close();
		}
	}

}

import java.util.Scanner;

		Scanner scan = new Scanner(System.in);

		for(int i=1; i<=T; i++) {
			int[] time = new int[4];
			int hour = 0;
			int minute = 0;
			
			for(int j=0; j<time.length; j++) {
				time[j] = scan.nextInt();
			}
			hour = time[0] + time[2];
			minute = time[1] + time[3];
			if(minute >=60) {
				minute -= 60;
				hour += 1;
			}
			if(hour >=13)	hour -= 12;
			System.out.format("#%d %d %d\n", i, hour, minute);
		}
		
	}

}
